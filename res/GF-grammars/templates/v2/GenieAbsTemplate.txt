abstract {abs_grammar_name} = {{
  flags coding = utf8 ;
  flags startcat = Triplet ;
  -- [s] John [r] lives in [o] New York [e] [s] John [r] likes [o] pizza [e] [s] Mary [r] lives in [o] London [e] --
  cat
    Triplet ; Subject ; Relation; Object ; Triplet_ending_marker; Subject_marker ; Relation_marker; Object_marker ; Rel ; Entity ;
  fun
    Repeat:  Triplet -> Triplet -> Triplet ;
    Empty: Triplet;
    Build_triplet : Subject -> Relation -> Object -> Triplet_ending_marker -> Triplet ;
    Build_subject: Subject_marker -> Entity -> Subject;
    Build_relation: Relation_marker -> Rel -> Relation;
    Build_object: Object_marker -> Entity -> Object;

    Build_subject_marker: Subject_marker;
    Build_relation_marker: Relation_marker;
    Build_object_marker: Object_marker;
    Build_triplet_ending_marker: Triplet_ending_marker;

    -- the following should be automatically generated --
    {entities_str}: Entity; -- line to replace, "Germany, France, UK, US" --
    {relations_str}: Rel; -- line to replace, "Is, Has" --
}}
